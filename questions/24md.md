```
/**
 * @param {number[]} nums
 * @return {void} Do not return anything, modify nums in-place instead.
 */
var moveZeroes = function(nums) {
    var index = 0;
    for(var i = 0 ; i < nums.length ; i++){
        var n = nums[i]; 
        // not zero, index++, push to array
        if(n !== 0){
            nums[index++] = n;    
        }
    }
    
    // after index to zero
    for(index ; index < nums.length ; index++){
        nums[index] = 0;
    }
};
```
